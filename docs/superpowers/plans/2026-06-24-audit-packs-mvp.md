# Audit Packs MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
> **Canonical save location (once out of plan mode):** `docs/superpowers/plans/2026-06-24-audit-packs-mvp.md`

**Goal:** Build an open-source GitHub Action that runs Checkov on changed IaC in a PR, maps findings onto NIST 800-53 controls (with a SOC 2 crosswalk), proves a hybrid path with one authored Semgrep rule, and posts unified, control-tagged, evidence-backed inline PR comments with a severity gate.

**Architecture:** A Python orchestrator (packaged as a Docker GitHub Action) runs detection engines (Checkov; one custom Semgrep rule), normalizes all output to a common SARIF-derived `Finding` model, filters to PR-changed lines, maps each finding to framework controls via YAML "packs" (NIST 800-53 canonical + SOC 2 crosswalk), and reports control-tagged findings as inline PR review comments plus a summary check-run. Detection is delegated to existing engines; the product value is the control-mapping + evidence + PR UX.

**Tech Stack:** Python 3.11, PyYAML, Checkov (invoked as CLI → SARIF), Semgrep CLI (→ SARIF), GitHub REST API via `requests`, pytest, Docker, `action.yml` (Docker action).

## Global Constraints

- Python version floor: **3.11** (uses `X | Y` type syntax, `tomllib` not required).
- Engines are invoked as **subprocesses producing SARIF**; never re-implement detection logic.
- Findings reported on a PR are **diff-filtered**: only findings on lines added/changed in the PR.
- Packs are **data, not code**: a framework pack = control catalog + mappings to engine check IDs.
- License: **Apache-2.0**. No paid-tier engine features (Semgrep Pro / Bearer Pro) may be required.
- Severity vocabulary is exactly: `low`, `medium`, `high`, `critical`.

# src/audit_packs/diff.py
import re

_HUNK = re.compile(r"^@@ -\d+(?:,\d+)? \+(\d+)(?:,(\d+))? @@")

def parse_unified_diff(diff_text: str) -> dict[str, set[int]]:
    result: dict[str, set[int]] = {}
    current: str | None = None
    for line in diff_text.splitlines():
        if line.startswith("+++ b/"):
            current = line[len("+++ b/"):].strip()
            continue
        if line.startswith("+++ ") or line.startswith("--- "):
            continue
        m = _HUNK.match(line)
        if m and current is not None:
            start = int(m.group(1))
            count = int(m.group(2)) if m.group(2) is not None else 1
            if count > 0:
                result.setdefault(current, set()).update(range(start, start + count))
    return {f: lines for f, lines in result.items() if lines}
```

# src/audit_packs/normalize.py
from audit_packs.models import Finding

_LEVEL_TO_SEVERITY = {"error": "high", "warning": "medium", "note": "low", "none": "low"}

def sarif_to_findings(sarif: dict, engine: str) -> list[Finding]:
    findings: list[Finding] = []
    for run in sarif.get("runs", []):
        for res in run.get("results", []):
            locs = res.get("locations", [])
            if not locs:
                continue
            phys = locs[0].get("physicalLocation", {})
            path = phys.get("artifactLocation", {}).get("uri", "")
            line = phys.get("region", {}).get("startLine", 1)
            msg = res.get("message", {}).get("text", "")
            severity = _LEVEL_TO_SEVERITY.get(res.get("level", "warning"), "medium")
            findings.append(Finding(
                check_id=res.get("ruleId", ""),
                engine=engine,
                file=path,
                line=int(line),
                severity=severity,
                message=msg,
                evidence=msg,
            ))
    return findings
```

# src/audit_packs/packs.py
import os
import yaml
from audit_packs.models import Finding, ControlFinding

def load_pack(path: str) -> dict:
    with open(path) as fh:
        data = yaml.safe_load(fh) or {}
    if "id" not in data or "controls" not in data:
        raise ValueError(f"pack {path} missing required keys 'id'/'controls'")
    return data

def _pack_path(packs_dir: str, pack_id: str) -> str:
    return os.path.join(packs_dir, f"{pack_id}.yaml")

def _canonical_index(pack: dict) -> dict[tuple[str, str], tuple[str, str]]:
    """(engine, check_id) -> (control_id, control_title)"""
    index: dict[tuple[str, str], tuple[str, str]] = {}
    for control in pack["controls"]:
        for group in control.get("checks", []):
            engine = group["engine"]
            for cid in group["ids"]:
                index[(engine, cid)] = (control["id"], control.get("title", control["id"]))
    return index

def map_findings(findings: list[Finding], packs_dir: str, frameworks: list[str]) -> list[ControlFinding]:
    results: list[ControlFinding] = []
    for fw in frameworks:
        pack = load_pack(_pack_path(packs_dir, fw))
        crosswalk_id = pack.get("crosswalk")
        canonical = load_pack(_pack_path(packs_dir, crosswalk_id)) if crosswalk_id else pack
        check_index = _canonical_index(canonical)
        if crosswalk_id:
            # map canonical control id -> this framework's (control_id, title)
            cw: dict[str, tuple[str, str]] = {}
            for control in pack["controls"]:
                for mapped in control.get("maps_to", []):
                    cw[mapped] = (control["id"], control.get("title", control["id"]))
        for f in findings:
            hit = check_index.get((f.engine, f.check_id))
            if not hit:
                continue
            canonical_control_id, canonical_title = hit
            if crosswalk_id:
                if canonical_control_id not in cw:
                    continue
                control_id, title = cw[canonical_control_id]
            else:
                control_id, title = canonical_control_id, canonical_title
            results.append(ControlFinding(f, fw, control_id, title))
    return results
```

# src/audit_packs/engines.py
import json
import subprocess

def run_checkov(target_dir: str) -> dict:
    proc = subprocess.run(
        ["checkov", "-d", target_dir, "--output", "sarif", "--output-file-path", "console"],
        capture_output=True, text=True, check=False,
    )
    # checkov prints SARIF to stdout when output-file-path is "console"
    try:
        return json.loads(proc.stdout)
    except json.JSONDecodeError:
        return {"runs": []}

def run_semgrep(target_dir: str, rules_path: str) -> dict:
    proc = subprocess.run(
        ["semgrep", "scan", "--config", rules_path, "--sarif", target_dir],
        capture_output=True, text=True, check=False,
    )
    try:
        return json.loads(proc.stdout)
    except json.JSONDecodeError:
        return {"runs": []}
```

import requests
from audit_packs.models import ControlFinding, severity_rank

def build_comments(control_findings: list[ControlFinding], commit_sha: str) -> list[dict]:
    comments = []
    for cf in control_findings:
        f = cf.finding
        body = (
            f"**Compliance control touched: `{cf.framework}` / {cf.control_id} — {cf.control_title}**\n\n"
            f"- Severity: `{f.severity}`\n"
            f"- Engine: `{f.engine}` (`{f.check_id}`)\n"
            f"- Finding: {f.message}\n\n"
            f"Evidence:\n```\n{f.evidence}\n```"
        )
        comments.append({"path": f.file, "line": f.line, "side": "RIGHT", "body": body})
    return comments

def gate_failed(control_findings: list[ControlFinding], fail_on: str) -> bool:
    threshold = severity_rank(fail_on)
    return any(severity_rank(cf.finding.severity) >= threshold for cf in control_findings)

def post_review(comments, summary, *, repo, pr_number, token, commit_sha) -> None:
    url = f"https://api.github.com/repos/{repo}/pulls/{pr_number}/reviews"
    headers = {"Authorization": f"Bearer {token}", "Accept": "application/vnd.github+json"}
    payload = {
        "commit_id": commit_sha,
        "body": summary,
        "event": "COMMENT",
        "comments": comments,
    }
    resp = requests.post(url, json=payload, headers=headers, timeout=30)
    resp.raise_for_status()

```python
# src/audit_packs/cli.py
import os
import subprocess
import sys
from audit_packs.engines import run_checkov, run_semgrep
from audit_packs.normalize import sarif_to_findings
from audit_packs.diff import parse_unified_diff
from audit_packs.packs import map_findings
from audit_packs.report import build_comments, gate_failed, post_review

def analyze(repo_dir, changed, packs_dir, rules_path, frameworks):
    findings = []
    findings += sarif_to_findings(run_checkov(repo_dir), "checkov")
    findings += sarif_to_findings(run_semgrep(repo_dir, rules_path), "semgrep")
    in_diff = [f for f in findings if f.line in changed.get(os.path.basename(f.file), changed.get(f.file, set()))]
    return map_findings(in_diff, packs_dir, frameworks)

def main() -> int:
    repo = os.environ["GITHUB_REPOSITORY"]
    token = os.environ["GITHUB_TOKEN"]
    pr_number = os.environ["PR_NUMBER"]
    base_ref = os.environ.get("BASE_REF", "origin/main")
    commit_sha = os.environ["GITHUB_SHA"]
    frameworks = [s.strip() for s in os.environ.get("FRAMEWORKS", "nist-800-53").split(",") if s.strip()]
    fail_on = os.environ.get("FAIL_ON", "high")
    workspace = os.environ.get("GITHUB_WORKSPACE", ".")
    packs_dir = os.environ.get("PACKS_DIR", "/app/packs")
    rules_path = os.environ.get("RULES_PATH", "/app/rules/weak-cipher.yaml")

    diff_text = subprocess.run(
        ["git", "diff", "--unified=0", f"{base_ref}...HEAD"],
        cwd=workspace, capture_output=True, text=True, check=True,
    ).stdout
    changed = parse_unified_diff(diff_text)

    cfs = analyze(workspace, changed, packs_dir, rules_path, frameworks)
    comments = build_comments(cfs, commit_sha)
    summary = f"Audit Packs: {len(cfs)} control-tagged finding(s) across {', '.join(frameworks)}."
    if comments:
        post_review(comments, summary, repo=repo, pr_number=pr_number, token=token, commit_sha=commit_sha)
    return 1 if gate_failed(cfs, fail_on) else 0

if __name__ == "__main__":
    sys.exit(main())
```

## Background & rationale (condensed feasibility review)

Detection is already solved by OSS engines (Checkov for IaC; Bearer/Privado for app-code PII/GDPR; Semgrep taint). Cross-file dataflow — the hard part — is gated behind paid tiers (Semgrep Pro, Bearer Pro), so re-building it is a multi-engineer-year trap. SOC 2 / ISO 27001 / NIST 800-53 / FedRAMP are mostly governance controls; the code-observable slice is the *same* technical checks, with NIST 800-53 as the shared substrate others crosswalk onto. Therefore a per-framework pack is a control-**mapping**, not a detector, and the defensible product is the **aggregator + OSCAL-style mapping + evidence-backed PR UX** (hybrid: also author rules for gaps). The strongest future differentiator — deferred — is an AI verifier→adversarial→judge pass to suppress false positives, the least-commoditized OSS piece.
